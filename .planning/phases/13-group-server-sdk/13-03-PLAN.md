**1. Create `GroupApi.kt`** following `AuthApi.kt` interface pattern exactly:
```kotlin
interface GroupApi {
    suspend fun createGroup(request: CreateGroupRequest): Either<AppError, GroupResponse>
    suspend fun getGroup(groupId: String): Either<AppError, GroupResponse>
    suspend fun updateGroup(groupId: String, request: UpdateGroupRequest): Either<AppError, GroupResponse>
    suspend fun deleteGroup(groupId: String): Either<AppError, Unit>
    suspend fun listAllGroups(): Either<AppError, List<GroupResponse>>
    suspend fun getMembers(groupId: String, cursor: String? = null, limit: Int = 20): Either<AppError, PaginatedMemberResponse>
    suspend fun addMember(groupId: String, request: AddMemberRequest): Either<AppError, MemberResponse>
    suspend fun removeMember(groupId: String, userId: String): Either<AppError, Unit>
    suspend fun registerMember(groupId: String, request: RegisterMemberRequest): Either<AppError, MemberResponse>
}
```

**2. Create `GroupApiImpl.kt`** following `AuthApiImpl.kt` implementation pattern:
- Constructor: `class GroupApiImpl(private val client: HttpClient) : GroupApi`
- Each method uses the `apiCall<T> { }` wrapper from the SDK module
- Use Ktor Resources plugin: `client.post(Groups.Create())`, `client.get(Groups.ById(groupId = groupId))`, etc.
- POST methods set `contentType(ContentType.Application.Json)` and `setBody(request)`
- GET methods with query params use the route class constructors (cursor/limit params are in the @Resource class)
- DELETE/remove endpoints use POST (matching route definitions)

Implementation for each method:
```kotlin
override suspend fun createGroup(request: CreateGroupRequest): Either<AppError, GroupResponse> =
    apiCall<GroupResponse> {
        client.post(Groups.Create()) {
            contentType(ContentType.Application.Json)
            setBody(request)
        }
    }

override suspend fun getGroup(groupId: String): Either<AppError, GroupResponse> =
    apiCall<GroupResponse> {
        client.get(Groups.ById(groupId = groupId))
    }

override suspend fun updateGroup(groupId: String, request: UpdateGroupRequest): Either<AppError, GroupResponse> =
    apiCall<GroupResponse> {
        client.post(Groups.Update(groupId = groupId)) {
            contentType(ContentType.Application.Json)
            setBody(request)
        }
    }

override suspend fun deleteGroup(groupId: String): Either<AppError, Unit> =
    apiCall<Unit> {
        client.post(Groups.Delete(groupId = groupId))
    }

override suspend fun listAllGroups(): Either<AppError, List<GroupResponse>> =
    apiCall<List<GroupResponse>> {
        client.get(Groups.ListAll())
    }

override suspend fun getMembers(groupId: String, cursor: String?, limit: Int): Either<AppError, PaginatedMemberResponse> =
    apiCall<PaginatedMemberResponse> {
        client.get(Groups.Members(groupId = groupId, cursor = cursor, limit = limit))
    }

override suspend fun addMember(groupId: String, request: AddMemberRequest): Either<AppError, MemberResponse> =
    apiCall<MemberResponse> {
        client.post(Groups.AddMember(groupId = groupId)) {
            contentType(ContentType.Application.Json)
            setBody(request)
        }
    }

override suspend fun removeMember(groupId: String, userId: String): Either<AppError, Unit> =
    apiCall<Unit> {
        client.post(Groups.RemoveMember(groupId = groupId, userId = userId))
    }

override suspend fun registerMember(groupId: String, request: RegisterMemberRequest): Either<AppError, MemberResponse> =
    apiCall<MemberResponse> {
        client.post(Groups.RegisterMember(groupId = groupId)) {
```

**1. Create `FakeGroupApiBuilder.kt`** following the existing FakeAuthApiBuilder/FakeUserApiBuilder pattern:
- Class with lambda properties for each GroupApi method, defaulting to `{ error(AppError.Server.Internal()) }`
- Method-specific DSL functions: `createGroup { ... }`, `getGroup { ... }`, etc.
- `build(): GroupApi` that creates an anonymous GroupApi implementation using the configured lambdas
- Example pattern:
```kotlin
class FakeGroupApiBuilder {
    var onCreateGroup: suspend (CreateGroupRequest) -> Either<AppError, GroupResponse> = { Either.Left(AppError.Server.Internal()) }
    var onGetGroup: suspend (String) -> Either<AppError, GroupResponse> = { Either.Left(AppError.Server.Internal()) }
    var onUpdateGroup: suspend (String, UpdateGroupRequest) -> Either<AppError, GroupResponse> = { _, _ -> Either.Left(AppError.Server.Internal()) }
    var onDeleteGroup: suspend (String) -> Either<AppError, Unit> = { Either.Left(AppError.Server.Internal()) }
    var onListAllGroups: suspend () -> Either<AppError, List<GroupResponse>> = { Either.Left(AppError.Server.Internal()) }
    var onGetMembers: suspend (String, String?, Int) -> Either<AppError, PaginatedMemberResponse> = { _, _, _ -> Either.Left(AppError.Server.Internal()) }
    var onAddMember: suspend (String, AddMemberRequest) -> Either<AppError, MemberResponse> = { _, _ -> Either.Left(AppError.Server.Internal()) }
    var onRemoveMember: suspend (String, String) -> Either<AppError, Unit> = { _, _ -> Either.Left(AppError.Server.Internal()) }
    var onRegisterMember: suspend (String, RegisterMemberRequest) -> Either<AppError, MemberResponse> = { _, _ -> Either.Left(AppError.Server.Internal()) }

    fun build(): GroupApi = object : GroupApi {
        override suspend fun createGroup(request: CreateGroupRequest) = onCreateGroup(request)
        override suspend fun getGroup(groupId: String) = onGetGroup(groupId)
        override suspend fun updateGroup(groupId: String, request: UpdateGroupRequest) = onUpdateGroup(groupId, request)
        override suspend fun deleteGroup(groupId: String) = onDeleteGroup(groupId)
        override suspend fun listAllGroups() = onListAllGroups()
        override suspend fun getMembers(groupId: String, cursor: String?, limit: Int) = onGetMembers(groupId, cursor, limit)
        override suspend fun addMember(groupId: String, request: AddMemberRequest) = onAddMember(groupId, request)
        override suspend fun removeMember(groupId: String, userId: String) = onRemoveMember(groupId, userId)
        override suspend fun registerMember(groupId: String, request: RegisterMemberRequest) = onRegisterMember(groupId, request)
    }
}
```
