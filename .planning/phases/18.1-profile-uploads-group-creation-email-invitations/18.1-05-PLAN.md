       package com.m2f.template.app.auth.invite
       
       import com.m2f.template.models.localization.StringKey
       
       data class InviteAcceptModel(
           val token: String = "",
           val isLoading: Boolean = true,
           val groupName: String = "",
           val inviterName: String = "",
           val isExpired: Boolean = false,
           val isAccepted: Boolean = false,
           val isAccepting: Boolean = false,
           val error: StringKey? = null,
           val acceptSuccess: Boolean = false,
           val joinedGroupSlug: String? = null,
       )
       
       sealed interface InviteAcceptIntent {
           data class LoadInvitation(val token: String) : InviteAcceptIntent
           object AcceptInvitation : InviteAcceptIntent

           data class SetInvitation(
               val token: String,
               val groupName: String,

               val isExpired: Boolean,
               val isAccepted: Boolean,
           ) : InviteAcceptMutation
           data class SetAccepting(val accepting: Boolean) : InviteAcceptMutation
           data class SetError(val error: StringKey?) : InviteAcceptMutation
           data class SetAcceptSuccess(val groupSlug: String) : InviteAcceptMutation
       }
       
       sealed interface InviteAcceptEvent {
           data class NavigateToGroup(val groupSlug: String) : InviteAcceptEvent
           object NavigateToLogin : InviteAcceptEvent

    3. Create InviteAcceptViewModel.kt:
       ```kotlin
       package com.m2f.template.app.auth.invite
       
       import androidx.lifecycle.viewModelScope
       import com.m2f.template.core.mvi.MviViewModel
       import com.m2f.template.models.localization.StringKey
       import com.m2f.template.sdk.Sdk
       import kotlinx.coroutines.flow.SharingStarted
       import kotlinx.coroutines.launch
       
       class InviteAcceptViewModel(
           private val sdk: Sdk,
       ) : MviViewModel<InviteAcceptIntent, InviteAcceptModel, InviteAcceptMutation, InviteAcceptEvent>(
           initialState = InviteAcceptModel(),
           modelSharingStarted = SharingStarted.Eagerly,
       ) {
           
           override fun take(intent: InviteAcceptIntent) {
               viewModelScope.launch {
                   when (intent) {
                       is InviteAcceptIntent.LoadInvitation -> handleLoadInvitation(intent.token)
                       is InviteAcceptIntent.AcceptInvitation -> handleAcceptInvitation()
                   }
               }
           }
           
           private suspend fun handleLoadInvitation(token: String) {
               sendMutation(InviteAcceptMutation.SetLoading(true))
               
               sdk.getInvitation(token).fold(
                   ifLeft = { error ->
                       sendMutation(InviteAcceptMutation.SetError(StringKey.fromCode(error.code) ?: StringKey.GENERIC_ERROR))
                       sendMutation(InviteAcceptMutation.SetLoading(false))
                   },
                   ifRight = { invitation ->
                       sendMutation(InviteAcceptMutation.SetInvitation(
                           token = invitation.token,
                           groupName = invitation.groupName,
                           inviterName = invitation.inviterName,
                           isExpired = invitation.isExpired,
                           isAccepted = invitation.isAccepted,
                       ))
                       sendMutation(InviteAcceptMutation.SetLoading(false))
                   },
               )
           }
           
           private suspend fun handleAcceptInvitation() {
               val current = model.value
               
               // Check if user is logged in
               if (!sdk.isLoggedIn()) {
                   // Redirect to login with return URL
                   sendEvent(InviteAcceptEvent.NavigateToLogin)
                   return
               }
               
               sendMutation(InviteAcceptMutation.SetAccepting(true))
               sendMutation(InviteAcceptMutation.SetError(null))
               
               sdk.acceptInvitation(current.token).fold(
                   ifLeft = { error ->
                       sendMutation(InviteAcceptMutation.SetError(StringKey.fromCode(error.code) ?: StringKey.GENERIC_ERROR))
                       sendMutation(InviteAcceptMutation.SetAccepting(false))
                   },
                   ifRight = { result ->
                       sendMutation(InviteAcceptMutation.SetAcceptSuccess(result.groupSlug))
                       sendEvent(InviteAcceptEvent.NavigateToGroup(result.groupSlug))
                   },
               )
           }
           
           override suspend fun reduce(model: InviteAcceptModel, mutation: InviteAcceptMutation): InviteAcceptModel =
               when (mutation) {
                   is InviteAcceptMutation.SetLoading -> model.copy(isLoading = mutation.loading)
                   is InviteAcceptMutation.SetInvitation -> model.copy(
                       token = mutation.token,
                       groupName = mutation.groupName,
                       inviterName = mutation.inviterName,
                       isExpired = mutation.isExpired,
                       isAccepted = mutation.isAccepted,
                   )
                   is InviteAcceptMutation.SetAccepting -> model.copy(isAccepting = mutation.accepting)
                   is InviteAcceptMutation.SetError -> model.copy(error = mutation.error)
                   is InviteAcceptMutation.SetAcceptSuccess -> model.copy(
                       acceptSuccess = true,
                       joinedGroupSlug = mutation.groupSlug,
                       isAccepting = false,
                   )
               }
       }
       ```
