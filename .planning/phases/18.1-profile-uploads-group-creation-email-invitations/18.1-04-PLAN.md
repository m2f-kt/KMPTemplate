       package com.m2f.server.groups.repository
       
       import com.m2f.server.groups.tables.InvitationsTable
       import kotlinx.datetime.Clock
       import kotlinx.datetime.Instant
       import org.jetbrains.exposed.sql.*
       import org.jetbrains.exposed.sql.transactions.experimental.newSuspendedTransaction
       import kotlin.uuid.Uuid
       
       data class InvitationRecord(
           val id: Uuid,
           val token: String,
           val groupId: Uuid,
           val email: String,
           val invitedBy: Uuid,
           val role: String,
           val expiresAt: Instant,
           val acceptedAt: Instant?,
           val acceptedBy: Uuid?,
           val createdAt: Instant,
       )
       
       class InvitationRepository {
           
           suspend fun create(
               groupId: Uuid,
               email: String,
               invitedBy: Uuid,
               role: String = "member",
               expiresAt: Instant,
           ): InvitationRecord = newSuspendedTransaction {
               val id = Uuid.random()
               val token = generateToken()
               val now = Clock.System.now()
               
               InvitationsTable.insert {
                   it[InvitationsTable.id] = id
                   it[InvitationsTable.token] = token
                   it[InvitationsTable.groupId] = groupId
                   it[InvitationsTable.email] = email
                   it[InvitationsTable.invitedBy] = invitedBy
                   it[InvitationsTable.role] = role
                   it[InvitationsTable.expiresAt] = expiresAt
                   it[InvitationsTable.createdAt] = now
               }
               
               InvitationRecord(id, token, groupId, email, invitedBy, role, expiresAt, null, null, now)
           }
           
           suspend fun findByToken(token: String): InvitationRecord? = newSuspendedTransaction {
               InvitationsTable.selectAll()
                   .where { InvitationsTable.token eq token }
                   .singleOrNull()
                   ?.toInvitationRecord()
           }
           
           suspend fun markAccepted(token: String, userId: Uuid): Boolean = newSuspendedTransaction {
               InvitationsTable.update({ InvitationsTable.token eq token }) {
                   it[acceptedAt] = Clock.System.now()
                   it[acceptedBy] = userId
               } > 0
           }
           
           private fun generateToken(): String {
               val chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789"
               return (1..32).map { chars.random() }.joinToString("")
           }
           
           private fun ResultRow.toInvitationRecord() = InvitationRecord(
               id = this[InvitationsTable.id],
               token = this[InvitationsTable.token],
               groupId = this[InvitationsTable.groupId],
               email = this[InvitationsTable.email],
               invitedBy = this[InvitationsTable.invitedBy],
               role = this[InvitationsTable.role],
               expiresAt = this[InvitationsTable.expiresAt],
               acceptedAt = this[InvitationsTable.acceptedAt],
               acceptedBy = this[InvitationsTable.acceptedBy],
               createdAt = this[InvitationsTable.createdAt],
           )
       }

    1. Create InvitationService.kt:
       ```kotlin
       package com.m2f.server.groups.service
       
       import arrow.core.Either
       import arrow.core.raise.either
       import com.m2f.server.auth.repository.UserRepository
       import com.m2f.server.auth.service.EmailService
       import com.m2f.server.groups.repository.GroupRepository
       import com.m2f.server.groups.repository.InvitationRepository
       import com.m2f.template.models.dto.AcceptInvitationResponse
       import com.m2f.template.models.dto.InvitationResponse
       import com.m2f.template.models.errors.AppError
       import kotlinx.datetime.Clock
       import kotlin.time.Duration.Companion.days
       import kotlin.uuid.Uuid
       
       class InvitationService(
           private val invitationRepository: InvitationRepository,
           private val groupRepository: GroupRepository,
           private val userRepository: UserRepository,
           private val groupService: GroupService,
           private val emailService: EmailService,
           private val baseUrl: String,  // From config, e.g., "https://myapp.com"
       ) {
           
           suspend fun createInvitation(
               groupId: Uuid,
               email: String,
               invitedBy: Uuid,
           ): Either<AppError, InvitationResponse> = either {
               // Verify inviter is admin of group
               val membership = groupRepository.findMembership(groupId, invitedBy)
               if (membership == null || membership.role !in listOf("admin", "owner")) {
                   raise(AppError.Forbidden("Only admins can invite"))
               }
               
               val expiresAt = Clock.System.now() + 7.days
               val record = invitationRepository.create(
                   groupId = groupId,
                   email = email,
                   invitedBy = invitedBy,
                   expiresAt = expiresAt,
               )
               
               val group = groupRepository.findById(groupId)
               val inviter = userRepository.findById(invitedBy)
               
               // Send invitation email
               val inviteUrl = "$baseUrl/invite?token=${record.token}"
               emailService.send(
                   to = email,
                   subject = "You've been invited to join ${group?.name ?: "a group"}",
                   body = """
                       Hi,
                       
                       ${inviter?.name ?: "Someone"} has invited you to join ${group?.name ?: "their group"}.
                       
                       Click here to accept: $inviteUrl
                       
                       This invitation expires in 7 days.
                   """.trimIndent(),
               )
               
               record.toResponse(group?.name ?: "", inviter?.name ?: "")
           }
           
           suspend fun getInvitation(token: String): Either<AppError, InvitationResponse> = either {
               val record = invitationRepository.findByToken(token)
                   ?: raise(AppError.NotFound("Invitation not found"))
               
               val group = groupRepository.findById(record.groupId)
               val inviter = userRepository.findById(record.invitedBy)
               
               record.toResponse(group?.name ?: "", inviter?.name ?: "")
           }
           
           suspend fun acceptInvitation(
               token: String,
               userId: Uuid,
           ): Either<AppError, AcceptInvitationResponse> = either {
               val record = invitationRepository.findByToken(token)
                   ?: raise(AppError.NotFound("Invitation not found"))
               
               if (record.acceptedAt != null) {
                   raise(AppError.Conflict("Invitation already accepted"))
               }
               
               if (record.expiresAt < Clock.System.now()) {
                   raise(AppError.Gone("Invitation expired"))
               }
               
               // Add user to group as member
               groupService.addMember(record.groupId, userId, record.role)
               
               // Mark invitation as accepted
               invitationRepository.markAccepted(token, userId)
               
               val group = groupRepository.findById(record.groupId)
               AcceptInvitationResponse(
                   groupId = record.groupId.toString(),
                   groupSlug = group?.slug ?: "",
                   role = record.role,
               )
           }
           
           private fun InvitationRecord.toResponse(groupName: String, inviterName: String) = InvitationResponse(
               id = id.toString(),
               token = token,
               groupId = groupId.toString(),
               groupName = groupName,
               email = email,
               inviterName = inviterName,
               role = role,
               expiresAt = expiresAt.toString(),
               isExpired = expiresAt < Clock.System.now(),
               isAccepted = acceptedAt != null,
           )
       }
       ```

           post<Invitations.Create> {
               val request = call.receive<CreateInvitationRequest>()

                   )
               }

       package com.m2f.template.sdk.api
       
       import arrow.core.Either
       import com.m2f.template.models.dto.*
       import com.m2f.template.models.errors.AppError
       import com.m2f.template.models.routes.Groups
       import com.m2f.template.models.routes.Invitations
       import io.ktor.client.*
       import io.ktor.client.request.*
       import io.ktor.resources.*
       
       class InvitationApiImpl(
           private val client: HttpClient,
       ) : InvitationApi {
           
           override suspend fun createInvitation(
               groupId: String,
               email: String,
           ): Either<AppError, InvitationResponse> = either {
               val result = client.post(
                   resourceHref(Groups.ById.Invitations.Create(
                       Groups.ById.Invitations(Groups.ById(Groups(), groupId))
                   ))
               ) {
                   setBody(CreateInvitationRequest(email))
               }
               handleResponse<InvitationResponse>(result)
           }
           
           override suspend fun getInvitation(token: String): Either<AppError, InvitationResponse> = either {
               val result = client.get(resourceHref(Invitations.ByToken(Invitations(), token)))
               handleResponse<InvitationResponse>(result)
           }
           
           override suspend fun acceptInvitation(token: String): Either<AppError, AcceptInvitationResponse> = either {
               val result = client.post(resourceHref(Invitations.Accept(Invitations()))) {
                   setBody(AcceptInvitationRequest(token))
               }
               handleResponse<AcceptInvitationResponse>(result)
           }
       }
