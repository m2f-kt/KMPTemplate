       fun TerminalAvatar(
           initials: String,
           modifier: Modifier = Modifier,
           imageUrl: String? = null,
           size: Dp = 36.dp,
       ) {
           // If imageUrl provided, try to load it
           if (imageUrl != null) {
               var showFallback by remember { mutableStateOf(false) }
               if (!showFallback) {
                   AsyncImage(
                       model = imageUrl,
                       contentDescription = "Avatar",
                       modifier = modifier.size(size).clip(RoundedCornerShape(50)),
                       contentScale = ContentScale.Crop,
                       onError = { showFallback = true },
                   )
                   return
               }
           }
           // Fallback to initials (existing code)
           Box(...) { BasicText(initials...) }
       }

    5. ImagePicker.android.kt: Use ActivityResultContracts.PickVisualMedia:
       ```kotlin
       @Composable
       actual fun rememberImagePickerLauncher(
           onImagePicked: (ImagePickerResult?) -> Unit,
       ): () -> Unit {
           val context = LocalContext.current
           val launcher = rememberLauncherForActivityResult(
               contract = ActivityResultContracts.PickVisualMedia(),
           ) { uri ->
               if (uri == null) {
                   onImagePicked(null)
                   return@rememberLauncherForActivityResult
               }
               val bytes = context.contentResolver.openInputStream(uri)?.readBytes() ?: return@rememberLauncherForActivityResult
               val mimeType = context.contentResolver.getType(uri) ?: "image/jpeg"
               onImagePicked(ImagePickerResult(bytes, "avatar.jpg", mimeType))
           }
           return { launcher.launch(PickVisualMediaRequest(ActivityResultContracts.PickVisualMedia.ImageOnly)) }
       }
       ```

           // Store bytes and show crop dialog
           sendMutation(ProfileMutation.ShowCropDialog(intent.bytes))
       }
       is ProfileIntent.CropConfirmed -> handleUploadAvatar()
       is ProfileIntent.CropCancelled -> sendMutation(ProfileMutation.HideCropDialog)
       
       private suspend fun handleUploadAvatar() {
           val current = model.value
           val bytes = current.pendingImageBytes ?: return
           sendMutation(ProfileMutation.SetUploadingAvatar(true))
           sendMutation(ProfileMutation.HideCropDialog)
           
           sdk.uploadAvatar(bytes, "avatar.jpg", "image/jpeg").fold(
               ifLeft = { error ->
                   sendMutation(ProfileMutation.SetServerError(StringKey.fromCode(error.code) ?: StringKey.GENERIC_ERROR))
               },
               ifRight = { user ->
                   sendMutation(ProfileMutation.SetAvatarUrl(user.avatarUrl))
               },
           )
           sendMutation(ProfileMutation.SetUploadingAvatar(false))
       }
