    **3. server/src/test/kotlin/com/m2f/template/AvatarRoutesTest.kt** -- Create the avatar round-trip test:

    ```kotlin
    @file:OptIn(ExperimentalUuidApi::class)
    package com.m2f.template

    class AvatarRoutesTest {
        companion object {
            private lateinit var database: R2dbcDatabase
            @BeforeClass @JvmStatic fun setup() { TestDatabase.start(); TestMinIO.start(); database = TestDatabase.createDatabase() }
            @AfterClass @JvmStatic fun teardown() { TestMinIO.stop(); TestDatabase.stop() }
        }

        @Test
        fun `avatar upload round-trip - upload image and verify avatarUrl in profile`() = runTest {
            avatarTestApp(database) {
                val userRepo = getKoin().get<UserRepository>()
                val s3Client = getKoin().get<S3Client>()
                val userId = createTestUser(userRepo, "avatar-roundtrip@test.com", "Avatar User", UserRole.User)
                val token = createTestToken(userId.toString(), UserRole.User)

                // Create the test bucket
                createTestBucket(s3Client, "test-uploads")

                // Prepare a small PNG-like image (1x1 pixel PNG)
                val pngBytes = byteArrayOf(
                    0x89.toByte(), 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A  // PNG magic bytes
                ) + ByteArray(100) // minimal padding

                val client = createClient { install(ContentNegotiation) { json() } }

                // Upload avatar via PUT /api/users/me/avatar
                val uploadResponse = client.put("/api/users/me/avatar") {
                    bearerAuth(token)
                    setBody(MultiPartFormDataContent(formData {
                        append("file", pngBytes, Headers.build {
                            append(HttpHeaders.ContentDisposition, "filename=\"avatar.png\"")
                            append(HttpHeaders.ContentType, "image/png")
                        })
                    }))
                }

                uploadResponse.status shouldBe HttpStatusCode.OK
                val userResponse = uploadResponse.body<UserResponse>()
                userResponse.avatarUrl shouldNotBe null
                userResponse.avatarUrl!! shouldContain "avatar"
                userResponse.email shouldBe "avatar-roundtrip@test.com"

                // Verify the avatarUrl was persisted in the database
                val updatedUser = userRepo.findById(userId)
                updatedUser shouldNotBe null
                updatedUser!!.avatarUrl shouldNotBe null
                updatedUser.avatarUrl shouldBe userResponse.avatarUrl
            }
        }
    }
    ```

    Use Kotest assertions (`shouldBe`, `shouldNotBe`, `shouldContain`) per codebase convention.
