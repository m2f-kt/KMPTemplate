    **2. Create EmailServiceTest.kt** (`server/auth/src/test/kotlin/.../EmailServiceTest.kt`):
    ```kotlin
    package com.m2f.server.auth

    import com.icegreen.greenmail.util.GreenMail
    import com.icegreen.greenmail.util.ServerSetup
    import com.m2f.core.config.configuration.Configuration
    import com.m2f.core.config.configuration.Env
    import com.m2f.server.auth.service.SmtpEmailService
    import io.kotest.matchers.shouldBe
    import io.kotest.matchers.string.shouldContain
    import kotlinx.coroutines.test.runTest
    import org.junit.After
    import org.junit.Before
    import org.junit.Test

    class EmailServiceTest {

        private lateinit var greenMail: GreenMail
        private lateinit var emailService: SmtpEmailService

        @Before
        fun setup() {
            // Start GreenMail on a random available port
            val smtpSetup = ServerSetup(0, null, ServerSetup.PROTOCOL_SMTP)
            greenMail = GreenMail(smtpSetup)
            greenMail.start()

            val port = greenMail.smtp.port

            val config = Configuration(
                env = Env(
                    email = Env.Email(
                        host = "localhost",
                        port = port,
                        username = "",
                        password = "",
                        fromAddress = "test@example.com",
                    ),
                ),
            )

            emailService = SmtpEmailService(config)
        }

        @After
        fun teardown() {
            greenMail.stop()
        }

        @Test
        fun `sendEmail delivers email to SMTP server`() = runTest {
            emailService.sendEmail(
                to = "recipient@test.com",
                subject = "Test Subject",
                body = "Hello from the test!",
            )

            val messages = greenMail.receivedMessages
            messages.size shouldBe 1

            val message = messages[0]
            message.subject shouldBe "Test Subject"
            message.allRecipients[0].toString() shouldBe "recipient@test.com"
            message.content.toString().trim() shouldContain "Hello from the test!"
        }

        @Test
        fun `sendEmail sets correct from address`() = runTest {
            emailService.sendEmail(
                to = "someone@test.com",
                subject = "From Test",
                body = "Checking from address",
            )

            val messages = greenMail.receivedMessages
            messages.size shouldBe 1
            messages[0].from[0].toString() shouldBe "test@example.com"
        }
    }
    ```

    Key points:
    - Uses `ServerSetup(0, ...)` to bind to a random free port (avoids port conflicts)
    - No authentication needed for GreenMail (matches MailHog behavior)
    - Each test method gets clean state via @Before/@After lifecycle
    - `greenMail.receivedMessages` gives access to all captured emails
    - Kotest `shouldBe` / `shouldContain` matchers for assertions
