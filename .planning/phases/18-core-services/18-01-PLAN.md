    **1. Create FileRoutes** (`server/files/.../routes/FileRoutes.kt`):
    ```kotlin
    fun Route.fileRoutes(fileService: FileService) {
        authenticate {
            post<Files.Upload> {
                conduitAuth { userId ->
                    val multipart = call.receiveMultipart()
                    var fileName = "unknown"
                    var contentType = "application/octet-stream"
                    var fileBytes: ByteArray? = null

                    multipart.forEachPart { part ->
                        when (part) {
                            is PartData.FileItem -> {
                                fileName = part.originalFileName ?: "unknown"
                                contentType = part.contentType?.toString() ?: "application/octet-stream"
                                fileBytes = part.provider().readRemaining().readByteArray()
                            }
                            else -> {}
                        }
                        part.dispose()
                    }

                    ensureNotNull(fileBytes) { InvalidContent }
                    fileService.upload(userId, fileName, contentType, fileBytes!!)
                }
            }

            get<Files.Get> { route ->
                conduitAuth { _ ->
                    val url = fileService.getPresignedUrl(route.fileKey)
                    mapOf("url" to url)
                }
            }
        }
    }
    ```
    Use `io.ktor.http.content.PartData`, `io.ktor.http.content.forEachPart`, `io.ktor.utils.io.readRemaining`, `io.ktor.utils.io.core.readByteArray`.
    The multipart handling follows Ktor 3.x patterns. Use `call.receiveMultipart()` inside the conduitAuth block (access `call` via RoutingContext).
    For the upload response, use `conduitAuth(HttpStatusCode.Created)` to return 201.
